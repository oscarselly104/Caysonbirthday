        // Fungsi Membuka Undangan

        function openInvitation() {

            // Sembunyikan Layar Welcome

            const welcomeScreen = document.getElementById('welcome-screen');

            welcomeScreen.style.opacity = '0';

           

            setTimeout(() => {

                welcomeScreen.classList.add('hidden');

               

                // Munculkan Konten Utama

                const mainContent = document.getElementById('main-content');

                mainContent.classList.remove('hidden');

               

                // Tampilkan Tombol Musik

                document.getElementById('music-btn').classList.remove('hidden');

                // Mainkan Musik

                const music = document.getElementById('bg-music');

                music.play().catch(e => console.log("Autoplay dicegah oleh browser."));

                // Trigger Animasi Fade-in (Efek Scroll)

                setTimeout(() => {

                    mainContent.style.opacity = '1';

                    const fadeElements = document.querySelectorAll('.fade-enter');

                    fadeElements.forEach((el, index) => {

                        setTimeout(() => {

                            el.classList.add('fade-enter-active');

                        }, index * 200); // Delay bertingkat

                    });

                }, 50);

            }, 1000); // Waktu transisi sesuai CSS duration-1000

        }

        // Kontrol Musik Latar

        let isPlaying = true;

        function toggleMusic() {

            const music = document.getElementById('bg-music');

            const icon = document.querySelector('#music-btn i');

           

            if (isPlaying) {

                music.pause();

                icon.classList.remove('fa-music', 'animate-pulse');

                icon.classList.add('fa-volume-mute');

            } else {

                music.play();

                icon.classList.remove('fa-volume-mute');

                icon.classList.add('fa-music', 'animate-pulse');

            }

            isPlaying = !isPlaying;

        }

        // Script Hitung Mundur (Countdown)

        // GANTI TANGGAL ACARA DI SINI (Format: Bulan Tanggal, Tahun Jam:Menit:Detik)

        // Contoh: "April 30, 2026 16:30:00"

        const countDownDate = new Date("April 30, 2026 16:30:00").getTime();

        const x = setInterval(function() {

            const now = new Date().getTime();

            const distance = countDownDate - now;

            const days = Math.floor(distance / (1000 * 60 * 60 * 24));

            const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));

            const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));

            const seconds = Math.floor((distance % (1000 * 60)) / 1000);

            document.getElementById("days").innerHTML = days < 10 ? "0" + days : days;

            document.getElementById("hours").innerHTML = hours < 10 ? "0" + hours : hours;

            document.getElementById("minutes").innerHTML = minutes < 10 ? "0" + minutes : minutes;

            document.getElementById("seconds").innerHTML = seconds < 10 ? "0" + seconds : seconds;

            if (distance < 0) {

                clearInterval(x);

                document.getElementById("days").innerHTML = "00";

                document.getElementById("hours").innerHTML = "00";

                document.getElementById("minutes").innerHTML = "00";

                document.getElementById("seconds").innerHTML = "00";

            }

        }, 1000);
